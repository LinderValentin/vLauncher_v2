        public void OpenGaming(object sender, RoutedEventArgs e)
        {

            Process.Start(new ProcessStartInfo("discord:") { UseShellExecute = true }); //Discord
            Process.Start(new ProcessStartInfo("steam:") { UseShellExecute = true });//Steam
            Process.Start(new ProcessStartInfo("modrinth:") { UseShellExecute = true });//Modrinth
        }
